===
env var in command
===
echo "home={{ HOME }}"
---
home={{ HOME }}

===
env var USER in both command and expected
===
echo "user={{ USER }}"
---
user={{ USER }}

===
PATH env var expands
===
echo "path={{ PATH }}" | cut -c1-10
---
path={{ PA }}
---
===
unknown env var left unchanged
===
echo '{{ NONEXISTENT_VAR_XYZ123 }}'
---
{{ NONEXISTENT_VAR_XYZ123 }}

===
multiple env vars
===
echo "{{ USER }}@{{ HOME }}"
---
{{ USER }}@{{ HOME }}

===
env var with FIXTURE_DIR
===
echo "user={{ USER }} dir=$CCTR_FIXTURE_DIR" | grep -o "user=[^ ]*"
---
user={{ USER }}
