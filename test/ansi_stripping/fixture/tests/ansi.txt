===
bold text ANSI codes are stripped
===
printf '\033[1mbold text\033[0m'
---
bold text

===
color codes are stripped
===
printf '\033[31mred\033[0m \033[32mgreen\033[0m \033[34mblue\033[0m'
---
red green blue

===
256-color codes are stripped
===
printf '\033[38;5;196mred256\033[0m'
---
red256

===
RGB color codes are stripped
===
printf '\033[38;2;255;0;0mrgb red\033[0m'
---
rgb red

===
cursor movement codes are stripped
===
printf 'hello\033[2Dworld'
---
helloworld

===
mixed formatting is stripped
===
printf '\033[1;4;31mbold underline red\033[0m normal'
---
bold underline red normal
