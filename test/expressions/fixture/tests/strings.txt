===
string equals
===
echo "msg: hello"
---
msg: {{ s }}
---
where
* s == "hello"

===
string not equals
===
echo "msg: hello"
---
msg: {{ s }}
---
where
* s != "world"
* s != ""

===
contains
===
echo "msg: hello world"
---
msg: {{ s }}
---
where
* s contains "world"
* s contains "hello"
* s contains " "

===
not contains (shorthand)
===
echo "msg: hello world"
---
msg: {{ s }}
---
where
* s not contains "foo"
* s not contains "bar"

===
not contains (long form)
===
echo "msg: hello world"
---
msg: {{ s }}
---
where
* not (s contains "foo")
* not (s contains "bar")

===
not startswith
===
echo "path: /usr/local/bin"
---
path: {{ p }}
---
where
* p not startswith "/home"
* p not startswith "/var"

===
not endswith
===
echo "file: document.pdf"
---
file: {{ f }}
---
where
* f not endswith ".txt"
* f not endswith ".doc"

===
not matches
===
echo "id: abc123"
---
id: {{ id }}
---
where
* id not matches /^[0-9]+$/
* id not matches /^[A-Z]+$/

===
startswith
===
echo "path: /usr/local/bin"
---
path: {{ p }}
---
where
* p startswith "/"
* p startswith "/usr"
* p startswith "/usr/local"

===
endswith
===
echo "file: document.pdf"
---
file: {{ f }}
---
where
* f endswith ".pdf"
* f endswith "pdf"
* f endswith "document.pdf"

===
matches regex simple
===
echo "id: abc123"
---
id: {{ id }}
---
where
* id matches /^[a-z]+[0-9]+$/

===
matches regex digits
===
echo "num: 12345"
---
num: {{ n: string }}
---
where
* n matches /^\d+$/
* n matches /^[0-9]{5}$/

===
matches regex alternation
===
echo "status: success"
---
status: {{ s }}
---
where
* s matches /^(success|failure|pending)$/

===
string with spaces
===
echo "message: hello beautiful world"
---
message: {{ m }}
---
where
* m contains "beautiful"
* m startswith "hello"
* m endswith "world"

===
combined string ops
===
echo "email: user@example.com"
---
email: {{ e }}
---
where
* e contains "@"
* e endswith ".com"
* e matches /^[^@]+@[^@]+\.[^@]+$/

===
strip leading whitespace
===
echo "s:   hello"
---
s: {{ s }}
---
where
* strip(s) == "hello"

===
strip trailing whitespace
===
echo "s: hello   "
---
s: {{ s }}
---
where
* strip(s) == "hello"

===
strip both ends
===
echo "s:   hello   "
---
s: {{ s }}
---
where
* strip(s) == "hello"

===
strip with tabs
===
printf "s:\t hello \t"
---
s: {{ s }}
---
where
* strip(s) == "hello"

===
strip preserves inner whitespace
===
echo "s:   hello world   "
---
s: {{ s }}
---
where
* strip(s) == "hello world"

===
strip on clean string
===
echo "s: hello"
---
s: {{ s }}
---
where
* strip(s) == "hello"

===
strip with literal string
===
echo "x"
---
x
---
where
* strip("  test  ") == "test"
