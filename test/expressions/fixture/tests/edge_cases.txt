%skip(Unix only) platform: windows
===
empty string
===
echo "val: "
---
val: {{ s }}
---
where
* s == ""
* len(s) == 0

===
string with spaces only
===
echo "val:    "
---
val: {{ s }}
---
where
* s == "   "
* len(s) == 3

===
deeply nested parentheses
===
echo "n: 10"
---
n: {{ n }}
---
where
* ((((((n)))))) == 10
* (((n + 5) * 2) - 10) / 2 == 10

===
zero
===
echo "n: 0"
---
n: {{ n }}
---
where
* n == 0
* n >= 0
* n <= 0
* n + 0 == 0
* n * 100 == 0

===
negative zero
===
echo "n: -0"
---
n: {{ n }}
---
where
* n == 0
* n == -0

===
very small decimal
===
echo "n: 0.0001"
---
n: {{ n }}
---
where
* n > 0
* n < 1
* n == 0.0001
* n * 10000 == 1

===
very large number
===
echo "n: 999999999"
---
n: {{ n }}
---
where
* n > 0
* n == 999999999
* n + 1 == 1000000000

===
negative decimal
===
echo "n: -3.14159"
---
n: {{ n }}
---
where
* n < 0
* n > -4
* -n > 3

===
string with double quotes escaped
===
echo 'val: say "hello"'
---
val: {{ s }}
---
where
* s contains "hello"
* len(s) == 11

===
string with backslash
===
printf '%s\n' 'path: C:\Users\test'
---
path: {{ p }}
---
where
* p contains "Users"
* p contains "test"
* p contains "\\"

===
unicode string
===
echo "msg: héllo wörld 你好"
---
msg: {{ m }}
---
where
* m contains "héllo"
* m contains "你好"

===
regex with special chars
===
echo "expr: (a+b)*c"
---
expr: {{ e }}
---
where
* e matches /^\(a\+b\)\*c$/

===
empty array membership
===
echo "n: 5"
---
n: {{ n }}
---
where
* not ([] contains n)

===
single element array
===
echo "n: 42"
---
n: {{ n }}
---
where
* [42] contains n

large array

===

---

===
power of zero
===
echo "n: 1"
---
n: {{ n }}
---
where
* n == 2 ^ 0
* n == 100 ^ 0
* n == 1 ^ 100

===
zero power
===
echo "n: 0"
---
n: {{ n }}
---
where
* n == 0 ^ 1
* n == 0 ^ 5

===
multiple not
===
echo "n: 5"
---
n: {{ n }}
---
where
* not not not not (n == 5)
* not not (n > 0)

===
comparison chain
===
echo "a: 1 b: 2 c: 3 d: 4 e: 5"
---
a: {{ a }} b: {{ b }} c: {{ c }} d: {{ d }} e: {{ e }}
---
where
* a < b and b < c and c < d and d < e
* e - a == 4
* a + b + c + d + e == 15

===
string equals with special chars
===
echo "s: !@#$%^&*()"
---
s: {{ s }}
---
where
* s == "!@#\$%^&*()"
* len(s) == 10
