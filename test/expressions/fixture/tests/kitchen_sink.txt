===
everything combined
===
echo "name: alice age: 30 score: 95.5 status: active"
---
name: {{ name }} age: {{ age }} score: {{ score }} status: {{ status }}
---
with
* name: string
* age: number
* score: number
* status: string
having
* len(name) == 5
* name startswith "a"
* name endswith "e"
* name contains "lic"
* age >= 18 and age < 100
* age == 30
* score > 90 and score <= 100
* score == 95.5
* ["active", "inactive"] contains status
* status == "active"
* not (age < 0)
* not (score < 0 or score > 100)
* (age > 20 and score > 90) or status == "admin"
* len(name) + age == 35
* age * 4 > score

===
nested parentheses
===
echo "x: 10"
---
x: {{ n }}
---
with
* n: number
having
* ((n > 5) and (n < 15)) or ((n > 100) and (n < 200))
* (((n + 5) * 2) - 10) == 20

===
all comparison operators
===
echo "v: 50"
---
v: {{ n }}
---
with
* n: number

* n == 50
* n != 49
* n < 51
* n <= 50
* n > 49
* n >= 50

===
all string operators
===
echo "s: hello_world_test"
---
s: {{ s }}
---
with
* s: string
having
* s == "hello_world_test"
* s != "other"
* s contains "world"
* s startswith "hello"
* s endswith "test"
* s matches /^[a-z_]+$/

===
multiple variables same type
===
echo "a: 10 b: 20 c: 30"
---
a: {{ a }} b: {{ b }} c: {{ c }}
---
with
* a: number
* b: number
* c: number
having
* a < b
* b < c
* a + b + c == 60
* a * b * c == 6000
* c - b == b - a
* (a + c) / 2 == b

===
boolean short circuit
===
echo "flag: 1"
---
flag: {{ n }}
---
with
* n: number
having
* n == 1 or (1 / 0 == 0)
* not (n == 0 and (1 / 0 == 0))
