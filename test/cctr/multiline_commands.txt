===
variable assignment and echo
===
x=hello; echo $x
---
hello

===
multiple statements with semicolons
===
a=1; b=2; echo $((a + b))
---
3

===
command substitution
===
x=$(echo hello); echo "$x world"
---
hello world

===
while loop counting
===
i=0; while [ $i -lt 3 ]; do echo $i; i=$((i + 1)); done
---
0
1
2

===
for loop over values
===
for x in a b c; do echo "item: $x"; done
---
item: a
item: b
item: c

===
if statement
===
x=5; if [ $x -gt 3 ]; then echo "big"; else echo "small"; fi
---
big

===
nested command substitution
===
echo "result: $(echo $(echo nested))"
---
result: nested

===
pipe chain
===
printf "c\na\nb\n" | sort | head -2
---
a
b

===
here string
===
cat <<< "hello from here string"
---
hello from here string

===
arithmetic expansion
===
echo $((2 * 3 + 4))
---
10

===
array-like iteration with seq
===
for i in $(seq 1 3); do printf "%d " $i; done; echo
---
1 2 3 

===
conditional with test
===
[ -d /tmp ] && echo "exists" || echo "missing"
---
exists

===
multiline with backslash continuation in single command
===
echo "line1" && \
echo "line2" && \
echo "line3"
---
line1
line2
line3

===
subshell grouping
===
(cd /tmp && pwd)
---
/tmp

===
brace expansion echo
===
echo {a,b,c}
---
a b c

===
environment variable in subshell
===
MY_VAR=test sh -c 'echo $MY_VAR'
---
test

===
read from process substitution
===
cat <(echo "from process sub")
---
from process sub

===
case statement
===
x=banana; case $x in apple) echo "fruit1";; banana) echo "fruit2";; esac
---
fruit2

===
function definition and call
===
greet() { echo "hello $1"; }; greet world
---
hello world
