%platform unix
===
json_string basic match
===
echo '"foo"'
---
{{ s: json string }}
---
where
* s == "foo"

===
json_string length
===
echo '"hello"'
---
{{ s: json string }}
---
where
* len(s) == 5

===
json_string with escapes
===
echo '"hello \"world\""'
---
{{ s: json string }}
---
where
* s contains "world"

===
json_string empty
===
echo '""'
---
{{ s: json string }}
---
where
* s == ""
* len(s) == 0

===
json_string with newline escape
===
printf '"line1\\nline2"'
---
{{ s: json string }}
---
where
* len(s) == 11

===
json_string with unicode
===
echo '"héllo wörld 你好"'
---
{{ s: json string }}
---
where
* s contains "héllo"

===
json_string type check
===
echo '"test"'
---
{{ s: json string }}
---
where
* type(s) == string

===
json_string in array context
===
echo '["a", "b", "c"]'
---
{{ arr: json array }}
---
where
* arr[0] == "a"
* type(arr[1]) == string

===
json_string comparison operators
===
echo '{"a": "apple", "b": "banana", "c": "cherry"}'
---
{{ o: json object }}
---
where
* o.a < o.b
* o.b < o.c
* o.c > o.a
* o.a <= o.a
* o.b >= o.b
* "aaa" < "aab"

===
json_string indexing
===
echo '{"s": "hello"}'
---
{{ o: json object }}
---
where
* o.s[0] == "h"
* o.s[1] == "e"
* o.s[4] == "o"
* o.s[-1] == "o"
* o.s[-2] == "l"

===
json_string indexing unicode
===
echo '{"s": "héllo"}'
---
{{ o: json object }}
---
where
* o.s[0] == "h"
* o.s[1] == "é"
* o.s[2] == "l"
* len(o.s) == 5

===
json_string lower upper functions
===
echo '{"s": "HeLLo WoRLd"}'
---
{{ o: json object }}
---
where
* lower(o.s) == "hello world"
* upper(o.s) == "HELLO WORLD"
* lower("ABC") == "abc"
* upper("xyz") == "XYZ"
