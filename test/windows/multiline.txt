%skip(Windows only) platform: not windows

===
two echo commands on separate lines
===
echo hello
echo world
---
hello
world

===
set variable then use it
===
set myvar=testing
echo %myvar%
---
testing

===
three commands
===
echo one
echo two
echo three
---
one
two
three

===
conditional if on multiple lines
===
set x=5
if %x%==5 echo yes
---
yes

===
for loop on multiple lines
===
for /L %%i in (1,1,3) do @echo %%i
---
1
2
3

===
create file read and delete
===
echo content> %TEMP%\cctr_multiline_test.txt
type %TEMP%\cctr_multiline_test.txt
del %TEMP%\cctr_multiline_test.txt
---
content

===
echo with variable capture
===
echo count: 42
echo done
---
count: {{ n }}
done
---
where
* n == 42

===
mixed commands with pipe
===
echo hello world
echo hello world| findstr hello
---
hello world
hello world
