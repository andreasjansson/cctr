%skip(Windows only) platform: not windows

===
multiple commands with ampersand
===
echo hello & echo world
---
hello
world

===
conditional execution with double ampersand
===
echo first && echo second
---
first
second

===
set and echo variable
===
set x=hello & echo %x%
---
hello

===
for loop counting
===
for /L %i in (1,1,3) do @echo %i
---
1
2
3

===
if statement true branch
===
if 1==1 (echo yes) else (echo no)
---
yes

===
if statement false branch
===
if 1==2 (echo yes) else (echo no)
---
no

===
echo with quotes
===
echo "hello world"
---
"hello world"

===
pipe to findstr
===
echo hello world | findstr hello
---
hello world

===
arithmetic with set /a
===
set /a result=2+3 & echo %result%
---
5

===
multiple lines with caret continuation
===
echo line1 ^
& echo line2
---
line1
line2

===
dir with /b flag (basic format)
===
echo test > %TEMP%\cctr_test.txt & dir /b %TEMP%\cctr_test.txt & del %TEMP%\cctr_test.txt
---
cctr_test.txt
