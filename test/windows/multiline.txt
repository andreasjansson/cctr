%platform windows

===
two echo commands on separate lines
===
Write-Output "hello"
Write-Output "world"
---
hello
world

===
set variable then use it
===
$myvar = "testing"
Write-Output $myvar
---
testing

===
three commands
===
Write-Output "one"
Write-Output "two"
Write-Output "three"
---
one
two
three

===
conditional if on multiple lines
===
$x = 5
if ($x -eq 5) { Write-Output "yes" }
---
yes

===
for loop on multiple lines
===
for ($i = 1; $i -le 3; $i++) { Write-Output $i }
---
1
2
3

===
echo with variable capture
===
Write-Output "count: 42"
Write-Output "done"
---
count: {{ n }}
done
---
where
* n == 42

===
mixed commands with pipe
===
Write-Output "hello world"
Write-Output "goodbye world" | ForEach-Object { $_.ToUpper() }
---
hello world
GOODBYE WORLD
