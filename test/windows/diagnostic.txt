%platform windows

===
show OS environment variable
===
Write-Output "OS=$env:OS"
---
OS={{ os }}

===
show powershell version
===
$PSVersionTable.PSVersion.Major
---
{{ version }}
---
where
* version >= 5

===
show what shell we are using via COMSPEC
===
Write-Output $env:COMSPEC
---
{{ shell }}
---
where
* shell contains "cmd.exe"
