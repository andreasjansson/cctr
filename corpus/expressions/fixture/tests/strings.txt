===
string equals
===
echo "msg: hello"
---
msg: {{ s }}
---
with
* s: string
having
* s == "hello"

===
string not equals
===
echo "msg: hello"
---
msg: {{ s }}
---
with
* s: string
having
* s != "world"
* s != ""

===
contains
===
echo "msg: hello world"
---
msg: {{ s }}
---
with
* s: string
having
* s contains "world"
* s contains "hello"
* s contains " "

===
not contains
===
echo "msg: hello world"
---
msg: {{ s }}
---
with
* s: string
having
* not (s contains "foo")
* not (s contains "bar")

===
startswith
===
echo "path: /usr/local/bin"
---
path: {{ p }}
---
with
* p: string
having
* p startswith "/"
* p startswith "/usr"
* p startswith "/usr/local"

===
endswith
===
echo "file: document.pdf"
---
file: {{ f }}
---
with
* f: string
having
* f endswith ".pdf"
* f endswith "pdf"
* f endswith "document.pdf"

===
matches regex simple
===
echo "id: abc123"
---
id: {{ id }}
---
with
* id: string
having
* id matches /^[a-z]+[0-9]+$/

===
matches regex digits
===
echo "num: 12345"
---
num: {{ n }}
---
with
* n: string
having
* n matches /^\d+$/
* n matches /^[0-9]{5}$/

===
matches regex alternation
===
echo "status: success"
---
status: {{ s }}
---
with
* s: string
having
* s matches /^(success|failure|pending)$/

===
string with spaces
===
echo "message: hello beautiful world"
---
message: {{ m }}
---
with
* m: string
having
* m contains "beautiful"
* m startswith "hello"
* m endswith "world"

===
combined string ops
===
echo "email: user@example.com"
---
email: {{ e }}
---
with
* e: string
having
* e contains "@"
* e endswith ".com"
* e matches /^[^@]+@[^@]+\.[^@]+$/
